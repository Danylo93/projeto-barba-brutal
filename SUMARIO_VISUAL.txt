💻 EXEMPLOS DE USO
═══════════════════════════════════════════════════════════════════════════

Usar um Hook:
─────────────
import { useAgendamentos } from '@/hooks/useAgendamentos';

const { agendamentos, loading, criar } = useAgendamentos();

Usar um Serviço:
────────────────
import agendamentoService from '@/services/api/agendamento.service';

const agendamentos = await agendamentoService.list();

Fazer Login:
────────────
import authService from '@/services/api/auth.service';

const response = await authService.loginUsuario(email, senha);
